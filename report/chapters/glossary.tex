\chapter*{Danh mục các từ viết tắt}
\addcontentsline{toc}{chapter}{Danh mục các từ viết tắt}

\begin{tabular}{|m{2.0cm}|m{5.6cm}|m{6.1cm}|}
\hline
\thead{Từ viết tắt} & \thead{Cụm từ đầy đủ} & \thead{Ý nghĩa / Mô tả} \\
\hline
API & Application Programming Interface & Tập hợp giao thức cho phép các phần mềm giao tiếp với nhau \\ \hline
APT & Advanced Persistent Threat & Chiến dịch tấn công có chủ đích \\ \hline
ASLR & Address Space Layout Randomization & Cơ chế bảo mật ngẫu nhiên hóa vị trí vùng nhớ khi tải tiến trình \\ \hline
ATT\&CK & Adversarial Tactics, Techniques, and Common Knowledge & Khung phân loại chiến thuật và kỹ thuật tấn công mạng do MITRE phát triển \\ \hline
AUC & Area Under the Curve & Chỉ số đánh giá mô hình phân loại, đo diện tích dưới đường cong ROC \\ \hline
AV & Antivirus & Phần mềm phát hiện và loại bỏ mã độc trên máy tính \\ \hline
COFF & Common Object File Format & Định dạng file nhị phân chuẩn dùng cho các module thực thi trên Windows \\ \hline
DEP & Data Execution Prevention & Cơ chế bảo mật ngăn thực thi mã từ vùng nhớ chỉ dành cho dữ liệu \\ \hline
DLL & Dynamic-Link Library & File thư viện chứa mã và tài nguyên dùng chung giữa các tiến trình Windows \\ \hline
EDR & Endpoint Detection and Response & Giải pháp giám sát, phát hiện và phản ứng với mối đe dọa trên thiết bị đầu cuối \\ \hline
EMBER & Endgame Malware BEnchmark for Research & Tập dữ liệu benchmark mã nguồn mở phục vụ nghiên cứu phát hiện mã độc PE \\ \hline
\end{tabular}





\begin{tabular}{|m{2.0cm}|m{5.6cm}|m{6.1cm}|}
\hline
\thead{Từ viết tắt} & \thead{Cụm từ đầy đủ} & \thead{Ý nghĩa / Mô tả} \\
\hline
EV & Extended Validation & Loại chứng chỉ SSL/TLS yêu cầu xác minh danh tính tổ chức nghiêm ngặt \\ \hline
FPR & False Positive Rate & Tỷ lệ mẫu lành tính bị mô hình phân loại nhầm thành mã độc \\ \hline
GUI & Graphical User Interface & Giao diện người dùng tương tác thông qua các thành phần đồ họa \\ \hline
IAT & Import Address Table & Bảng lưu địa chỉ các hàm được nạp động từ DLL khi thực thi \\ \hline
IOCTL & Input/Output Control & Cơ chế gửi lệnh điều khiển trực tiếp từ ứng dụng xuống driver kernel \\ \hline
IPC & Inter-Process Communication & Giao tiếp liên tiến trình \\ \hline
KMDF & Kernel-Mode Driver Framework & Framework của Microsoft hỗ trợ phát triển driver hoạt động ở chế độ nhân \\ \hline
LGBM & LightGBM & Thuật toán gradient boosting hiệu năng cao, tối ưu cho dữ liệu lớn \\ \hline
MBC & Malware Behavior Catalog & Danh mục chuẩn hóa các hành vi của mã độc phục vụ phân tích và phân loại \\ \hline
ML & Machine Learning & Lĩnh vực AI cho phép máy tính học từ dữ liệu mà không lập trình tường minh \\ \hline
PCA & Principal Component Analysis & Kỹ thuật giảm chiều dữ liệu bằng cách chiếu lên các thành phần chính \\ \hline
PE & Portable Executable & Định dạng file thực thi và thư viện chuẩn trên hệ điều hành Windows \\ \hline
\end{tabular}




\begin{tabular}{|m{2.0cm}|m{5.6cm}|m{6.1cm}|}
\hline
\thead{Từ viết tắt} & \thead{Cụm từ đầy đủ} & \thead{Ý nghĩa / Mô tả} \\
\hline
PID & Process Identifier & Số định danh duy nhất do hệ điều hành cấp cho mỗi tiến trình đang chạy \\ \hline
ROC & Receiver Operating Characteristic & Đồ thị biểu diễn sự đánh đổi giữa TPR và FPR của mô hình phân loại \\ \hline
ROP & Return-Oriented Programming & Kỹ thuật khai thác lỗ hổng bằng cách kết nối các đoạn mã có sẵn trong bộ nhớ \\ \hline
SHAP & SHapley Additive exPlanations & Phương pháp định lượng mức độ đóng góp của từng đặc trưng vào kết quả dự đoán \\ \hline
SVD & Singular Value Decomposition & Phép phân tích ma trận thành tích các ma trận đặc trưng, dùng trong giảm chiều dữ liệu \\ \hline
SVM & Support Vector Machine & Thuật toán phân loại tìm siêu phẳng tối ưu phân tách các lớp dữ liệu \\ \hline
TPM & Trusted Platform Module & Chip bảo mật phần cứng lưu trữ khóa mật mã và đảm bảo tính toàn vẹn hệ thống \\ \hline
WDK & Windows Driver Kit & Bộ công cụ và thư viện chính thức của Microsoft để phát triển driver Windows \\ \hline
WHQL & Windows Hardware Quality Labs & Chương trình kiểm định và cấp chứng nhận tương thích phần cứng của Microsoft \\ \hline
xAI & Explainable Artificial Intelligence & Hướng nghiên cứu AI tập trung vào khả năng giải thích quá trình ra quyết định của mô hình \\ \hline
\end{tabular}
